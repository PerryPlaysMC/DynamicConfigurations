```xml
<repositories>
  <repository>
      <id>jitpack.io</id>

  </repository>
</repositories>

<dependency>
    <groupId>com.github.PerryPlaysMC</groupId>
    <artifactId>DynamicConfigurations</artifactId>
    <version>1.0-SNAPSHOT</version>
</dependency>
```
